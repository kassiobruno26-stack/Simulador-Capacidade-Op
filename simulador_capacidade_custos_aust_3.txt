function save(){try{localStorage.setItem(STORE_KEY,JSON.stringify({state,uid}))}catch(e){}}

  return false;

function scenarioDefault(nc,ramp,churn,adjust){return {newClients:nc,ramp,churn,adjust}}
function normalize(){
  state.tasks=state.tasks||[];
  const pj=state.projection||{};
  const oldNC=pj.newClients||{advisor:0,ma:0,pontual:0};
  if(!pj.scenarios){
    pj.scenarios={
      conservador:scenarioDefault({advisor:0,ma:0,pontual:0},3,3,5),
      base:scenarioDefault({advisor:oldNC.advisor||1,ma:oldNC.ma||0,pontual:oldNC.pontual||0},2,2,5),
      agressivo:scenarioDefault({advisor:(oldNC.advisor||1)*2,ma:(oldNC.ma||0)*2||1,pontual:(oldNC.pontual||0)*2},1,1,5)
    };
  }
  ['conservador','base','agressivo'].forEach(k=>{const s=pj.scenarios[k];s.newClients=s.newClients||{advisor:0,ma:0,pontual:0};if(s.ramp==null)s.ramp=2;if(s.churn==null)s.churn=2;if(s.adjust==null)s.adjust=5});
  pj.horizon=pj.horizon||12;pj.hires=pj.hires||[];pj.activeScenario=pj.activeScenario||'base';
  delete pj.newClients;delete pj.fixedGrowth;
  state.projection=pj;
  state.people.forEach(p=>{
    if(!p.onbItems){
      p.onbItems=[];
      if(p.onboarding&&p.onboarding>0)p.onbItems.push({id:nid(),desc:'Equipamentos',value:p.onboarding,parcelas:p.onbParcelas||12});
    }
    delete p.onboarding;delete p.onbParcelas;
  });
  state.clients.forEach(c=>{if(!c.watch)c.watch={on:false,level:'media',reason:'',next:''};if(c.permuta==null)c.permuta=false});
}

function personMonthHours(p){return personWeekHours(p)*state.params.weeks}

function teamMonthHours(){return state.people.filter(p=>p.active).reduce((s,p)=>s+personMonthHours(p),0)}

function personWeekLoad(p){
  let l=0;
  if(p.hasManagement)l+=personWeekHours(p)*(p.mgmtSplit/100);
  state.clients.forEach(c=>{if(c.ownerId===p.id&&c.slot)l+=clientWeekHours(c)});
  state.tasks.forEach(t=>{if(t.ownerId===p.id&&t.slot)l+=taskWeekHours(t)});
  return l;
}
function personLoadSplit(p){
  const mgmt=p.hasManagement?personWeekHours(p)*(p.mgmtSplit/100):0;
  let cli=0,intr=0;
  state.clients.forEach(c=>{if(c.ownerId===p.id&&c.slot)cli+=clientWeekHours(c)});
  state.tasks.forEach(t=>{if(t.ownerId===p.id&&t.slot)intr+=taskWeekHours(t)});
  return {mgmt,cli,intr};
}
function personOccupation(p){const cap=personWeekHours(p);return cap>0?(personWeekLoad(p)/cap)*100:0}

function freeSlackClients(p){
  const cap=personWeekHours(p);const free=cap-personWeekLoad(p);
  const owned=state.clients.filter(c=>c.ownerId===p.id&&c.slot);
  let avg=owned.length?owned.reduce((s,c)=>s+clientWeekHours(c),0)/owned.length:(1*(state.params.attendance/100)+3);
  return avg>0?Math.max(0,Math.floor(free/avg)):0;
}

function avgHoursType(t){
  const cs=state.clients.filter(c=>c.type===t);
  if(cs.length)return cs.reduce((s,c)=>s+clientMonthHours(c),0)/cs.length;
  const d={advisor:[1,3],ma:[1.5,6],pontual:[1,4]}[t]||[1,3];
  return (d[0]*(state.params.attendance/100)+d[1])*state.params.weeks;
}
// simula um cenário mês a mês com ramp-up, churn, reajuste anual
function simulate(scenKey){
  const pj=state.projection;const s=pj.scenarios[scenKey];const months=[];
  const allocated=state.clients.filter(c=>c.slot);
  const baseRevenue=allocated.reduce((sum,c)=>sum+clientRevenue(c),0);
  const basePeople=state.people.filter(p=>p.active);
  const basePeopleCost=basePeople.reduce((sum,p)=>sum+personMonthlyCost(p),0);
  const baseCapacity=basePeople.reduce((sum,p)=>sum+personMonthHours(p),0);
  let baseDemand=0;allocated.forEach(c=>baseDemand+=clientMonthHours(c));state.tasks.filter(t=>t.slot).forEach(t=>baseDemand+=taskWeekHours(t)*state.params.weeks);
  const newRevPerMonth=['advisor','ma','pontual'].reduce((a,t)=>a+(s.newClients[t]||0)*(state.fees[t]||0),0);
  const newDemPerMonth=['advisor','ma','pontual'].reduce((a,t)=>a+(s.newClients[t]||0)*avgHoursType(t),0);
  const churn=(s.churn||0)/100;const ramp=Math.max(1,s.ramp||1);
  let cash=0,estouroMonth=null,paybackMonth=null;
  for(let m=1;m<=pj.horizon;m++){
    const adj=Math.pow(1+(s.adjust||0)/100,Math.floor((m-1)/12));
    // base cohort (start=1, ramp completo), churn a partir do mês seguinte
    let rev=baseRevenue*Math.pow(1-churn,m-1);
    let dem=baseDemand*Math.pow(1-churn,m-1);
    // cohorts de clientes novos: um por mês k=1..m
    for(let k=1;k<=m;k++){
      const age=m-k;const rf=Math.min(1,(age+1)/ramp);const surv=Math.pow(1-churn,age);
      rev+=newRevPerMonth*rf*surv;dem+=newDemPerMonth*rf*surv;
    }
    rev*=adj;
    // pessoas
    let peopleCost=basePeopleCost*adj,capacity=baseCapacity,headAdd=0;
    pj.hires.forEach(h=>{if(h.month<=m){peopleCost+=((h.salary||0)+(h.encargos||0))*adj;if(h.onbParcelas>0&&m<h.month+h.onbParcelas)peopleCost+=(h.onboarding||0)/h.onbParcelas;capacity+=(h.weekHours||0)*state.params.weeks;headAdd++}});
    const fixed=fixedTotal()*adj;
    const result=rev-peopleCost-fixed;cash+=result;
    const occ=capacity>0?(dem/capacity)*100:0;
    if(estouroMonth===null&&occ>100)estouroMonth=m;
    if(paybackMonth===null&&cash>=0&&result>=0)paybackMonth=m;
    months.push({m,revenue:rev,peopleCost,fixed,result,cash,occ,capacity,demand:dem,headcount:basePeople.length+headAdd});
  }
  return {months,estouroMonth,paybackMonth,total:months.reduce((a,x)=>a+x.result,0),finalCash:cash};
}
